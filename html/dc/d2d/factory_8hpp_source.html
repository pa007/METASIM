39 // The abstract factory itself.
40 // Implemented using the Singleton pattern
41 template <class manufacturedObj, typename classIDKey=defaultIDKeyType>
42 class genericFactory
43 {
44  // a BASE_CREATE_FN is a function that takes no parameters
45  // and returns an auto_ptr to a manufactuedObj. Note that
46  // we use no parameters, but you could add them
47  // easily enough to allow overloaded ctors, e.g.:
48  // typedef std::auto_ptr<manufacturedObj> (*BASE_CREATE_FN)(int);
49  typedef std::auto_ptr<manufacturedObj> (*BASE_CREATE_FN)(std::vector<std::string> &par);
50 
51  // FN_REGISTRY is the registry of all the BASE_CREATE_FN
52  // pointers registered. Functions are registered using the
53  // regCreateFn member function (see below).
54  typedef std::map<classIDKey, BASE_CREATE_FN> FN_REGISTRY;
55  FN_REGISTRY registry;
56 
57  // Singleton implementation - private ctor & copying, with
58  // no implementation on the copying.
59  genericFactory();
60  genericFactory(const genericFactory&); // Not implemented
61  genericFactory &operator=(const genericFactory&); // Not implemented
62  public:
63 
64  // Singleton access.
65  static genericFactory &instance();
66 
67  // Classes derived from manufacturedObj call this function once
68  // per program to register the class ID key, and a pointer to
69  // the function that creates the class.
70  void regCreateFn(const classIDKey &, BASE_CREATE_FN);
71 
72  // Create a new class of the type specified by className.
73  std::auto_ptr<manufacturedObj> create(const classIDKey &className, std::vector<std::string> &parms) const;
74 };

93 // Register the creation function. This simply associates the classIDKey
94 // with the function used to create the class. The return value is a dummy

103 // The create function simple looks up the class ID, and if it's in the list,
104 // the statement "(*i).second();" calls the function.
105 template <class manufacturedObj, typename classIDKey>
106 std::auto_ptr<manufacturedObj> genericFactory<manufacturedObj, classIDKey>::create(const classIDKey &className, std::vector<std::string> &parms) const
107 {
108  std::auto_ptr<manufacturedObj> ret(0);
109 
110  typename FN_REGISTRY::const_iterator regEntry=registry.find(className);
111  if (regEntry != registry.end()) {
112  return (*regEntry).second(parms);
113  }
114  return ret;
115 }
